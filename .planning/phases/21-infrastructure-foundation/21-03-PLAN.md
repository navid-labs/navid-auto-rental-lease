2. Add a `coverage` block to the `test` object in `vitest.config.mts`. The full updated config should be:
```typescript
import { defineConfig } from 'vitest/config'
import react from '@vitejs/plugin-react'
import { fileURLToPath } from 'url'
import path from 'path'

const __dirname = path.dirname(fileURLToPath(import.meta.url))

export default defineConfig({
  plugins: [react()],
  test: {
    environment: 'happy-dom',
    include: ['src/**/*.test.{ts,tsx}', 'tests/**/*.test.{ts,tsx}'],
    setupFiles: [],
    coverage: {
      provider: 'v8',
      include: ['src/**/*.{ts,tsx}'],
      exclude: [
        'src/**/*.test.{ts,tsx}',
        'src/**/*.d.ts',
        'src/types/**',
        'src/lib/api/generated/**',
        'src/app/**/layout.tsx',
        'src/app/**/loading.tsx',
        'src/app/**/error.tsx',
        'src/app/**/not-found.tsx',
      ],
      reporter: ['text', 'text-summary', 'html'],
      reportsDirectory: './coverage',
    },
  },
  resolve: {
    alias: {
      '@': path.resolve(__dirname, './src'),
    },
  },
})
```

Key decisions:
- `provider: 'v8'` -- uses V8 engine native coverage (faster than Istanbul)
- `include: ['src/**/*.{ts,tsx}']` -- covers all source files (replaces removed `coverage.all` from Vitest 4)
- `exclude` -- skips test files, type declarations, generated API code, and Next.js boilerplate files
- `reporter: ['text', 'text-summary', 'html']` -- text for terminal, html for detailed browsing
- `reportsDirectory: './coverage'` -- standard location (already in .gitignore)
- NO `thresholds` -- per user decision, measure baseline only, do not enforce yet

3. Verify `./coverage` is in `.gitignore`. If not, add it.

4. Run `bun run test:coverage` to produce the baseline coverage report and verify the output includes statements, branches, functions, and lines percentages.
